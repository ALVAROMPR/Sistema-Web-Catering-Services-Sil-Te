CREATE DATABASE sistemacatering_db;

-- ROLES
CREATE TABLE roles (
  id_rol SERIAL PRIMARY KEY,
  nombre_rol VARCHAR(50) UNIQUE NOT NULL
);

-- PERMISOS
CREATE TABLE permisos (
  id_permiso SERIAL PRIMARY KEY,
  nombre VARCHAR(100),
  descripcion TEXT
);

-- ROLES_PERMISOS
CREATE TABLE roles_permisos (
  id_rol INT REFERENCES roles(id_rol),
  id_permiso INT REFERENCES permisos(id_permiso),
  PRIMARY KEY (id_rol, id_permiso)
);

-- USUARIOS
CREATE TABLE usuarios (
  id_usuario SERIAL PRIMARY KEY,
  username VARCHAR(50) UNIQUE NOT NULL,
  password TEXT NOT NULL,
  nombre_completo VARCHAR(100),
  id_rol INT REFERENCES roles(id_rol),
  estado BOOLEAN DEFAULT TRUE,
  fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- CARGOS
CREATE TABLE cargos (
  id_cargo SERIAL PRIMARY KEY,
  nombre_cargo VARCHAR(100)
);

-- CLIENTES
CREATE TABLE clientes (
  id_cliente SERIAL PRIMARY KEY,
  codigo_cliente VARCHAR(50) UNIQUE NOT NULL,
  nombre_completo VARCHAR(150),
  id_cargo INT REFERENCES cargos(id_cargo),
  estado BOOLEAN DEFAULT TRUE,
  creado_por INT REFERENCES usuarios(id_usuario),
  fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- CATEGORIAS
CREATE TABLE categorias (
  id_categoria SERIAL PRIMARY KEY,
  nombre VARCHAR(100)
);

-- UNIDADES DE MEDIDA
CREATE TABLE unidades_medida (
  id_unidad SERIAL PRIMARY KEY,
  nombre VARCHAR(50)
);

-- PRODUCTOS
CREATE TABLE productos (
  id_producto SERIAL PRIMARY KEY,
  nombre VARCHAR(150),
  precio NUMERIC(10,2),
  id_categoria INT REFERENCES categorias(id_categoria),
  id_unidad INT REFERENCES unidades_medida(id_unidad),
  stock INT DEFAULT 0,
  tiene_stock BOOLEAN DEFAULT TRUE,
  creado_por INT REFERENCES usuarios(id_usuario)
);

-- VENTAS
CREATE TABLE ventas (
  id_venta SERIAL PRIMARY KEY,
  id_cliente INT REFERENCES clientes(id_cliente),
  id_usuario INT REFERENCES usuarios(id_usuario),
  total NUMERIC(10,2),
  fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- DETALLE DE VENTA
CREATE TABLE detalle_venta (
  id_detalle SERIAL PRIMARY KEY,
  id_venta INT REFERENCES ventas(id_venta) ON DELETE CASCADE,
  id_producto INT REFERENCES productos(id_producto),
  cantidad INT,
  precio_unitario NUMERIC(10,2),
  subtotal NUMERIC(10,2)
);

-- TIPOS DE NOTIFICACION
CREATE TABLE tipos_notificacion (
  id_tipo SERIAL PRIMARY KEY,
  nombre VARCHAR(50)
);

-- NOTIFICACIONES
CREATE TABLE notificaciones (
  id_notificacion SERIAL PRIMARY KEY,
  id_tipo INT REFERENCES tipos_notificacion(id_tipo),
  mensaje TEXT,
  leido BOOLEAN DEFAULT FALSE,
  fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- AUDITORIA
CREATE TABLE auditoria (
  id_auditoria SERIAL PRIMARY KEY,
  id_usuario INT REFERENCES usuarios(id_usuario),
  accion VARCHAR(100),
  tabla_afectada VARCHAR(100),
  descripcion TEXT,
  fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- BUSQUEDAS
CREATE TABLE busquedas_cliente (
  id_busqueda SERIAL PRIMARY KEY,
  id_usuario INT REFERENCES usuarios(id_usuario),
  termino_busqueda TEXT,
  fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- HISTORIAL DE CONSUMO
CREATE TABLE historial_consumo (
  id_historial SERIAL PRIMARY KEY,
  id_cliente INT REFERENCES clientes(id_cliente),
  id_producto INT REFERENCES productos(id_producto),
  frecuencia INT,
  ultima_compra TIMESTAMP
);
